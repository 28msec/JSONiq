5.7. JSON representation of the Object Schema Type against which the JSON representations of all Object Schema Types are valid.

{
  "$name" : "object-schema-type",
  {
    "$object" : {
      "$$name" : {
        "$type" : "string",
        "$optional" : true
      },
      "$$object" : {
        "$type" : "#object-layout"
      },
      "$$constraints" : {
        "$type" : { "$array" : [ "string" ] },
        "$optional" : true,
        "$default" : []
      },
    }
  }
}

{
  "$name" : "object-layout"
  "$object" : {
    "$any" : {
      "$type : {
        "$object" : {
          "$$type" : { "$type" : { "$union" : [ "string", "#schema-type" ] } },
          "$$optional" : { "$type" : "boolean", "$optional" : true, $default : "false" }
          "$$default" : { "$type" : "item", "$optional" : true, $default : "null" }
        }
      }
    }
  }
}
