3.9. JSON representation of the Object Schema Type against which the JSON representations of all Atomic Schema Types are valid.

 
{
  "$name" : "atomic-schema-type",
  "$union" : [
    "string",
    {
      "$object" : {
        "$$atomic" : { "$type" : "#atomic-schema-type" },
        "$$name" : { "$type" : "string",
                     "$optional" : true },
        "$$length" : { "$type" : "integer",
                       "$optional" : true },
        "$$minLength" : { "$type" : "integer",
                          "$optional" : true },
        "$$maxLength" : { "$type" : "integer",
                          "$optional" : true },
        "$$totalDigits" : { "$type" : "integer",
                            "$optional" : true },
        "$$fractionDigits" : { "$type" : "integer",
                               "$optional" : true },
        "$$pattern" : { "$type" : "string",
                        "$optional" : true },
        "$$enumeration" : { "$type" : { "$array" : [ "anyAtomicType" ] },
                            "$optional" : true },
        "$$whitespace" : { "$type" : { "$atomic" : "string",
                                       "$enum" : [ "collapse", "preserve", "replace" ] },
                           "$optional" : true },
        "$$maxInclusie" : { "$type" : "anyAtomicType",
                            "$optional" : true },
        "$$maxExclusie" : { "$type" : "anyAtomicType",
                            "$optional" : true },
        "$$minExclusie" : { "$type" : "anyAtomicType",
                            "$optional" : true },
        "$$minInclusie" : { "$type" : "anyAtomicType",
                            "$optional" : true },
        "$$explicitTimezone" : { "$type" : { "$atomic" : "string",
                                 "$enum" : [ "required", "prohibited", "optional" ] },
                                 "$optional" : true },
        "$$constraints" : { "$type" : { "$array" : [ "string" ],
                            "$optional" : true },
      }
    }
  ]
}
