3.7. JSON representation of the Object Schema Type against which the JSON representations of all Atomic Schema Types are valid.

 
{
  "$name" : "atomic-schema-type",
  "$object" : {
    "$$atomic" : { "$type" : "string" },
    "$$name" : { "$type" : "string",
                 "$default" : "" },
    "$$length" : { "$type" : { "$union" : [ "integer", "null" ] } }
                   "$default" : null },
    "$$minLength" : { "$type" : { "$union" : [ "integer", "null" ] } }
                      "$default" : null },
    "$$maxLength" : { "$type" : { "$union" : [ "integer", "null" ] } }
                      "$default" : null },
    "$$totalDigits" : { "$type" : { "$union" : [ "integer", "null" ] } }
                        "$default" : null },
    "$$fractionDigits" : { "$type" : { "$union" : [ "integer", "null" ] } }
                           "$default" : null },
    "$$pattern" : { "$type" : { "$union" : [ "string", "null" ] } }
                    "$default" : null },
    "$$enumeration" : { "$type" : { "$array" : [ "atomic" ] },
                        "$default" : [] },
    "$$maxInclusie" : { "$type" : "atomic",
                        "$default" : null },
    "$$maxExclusie" : { "$type" : "atomic",
                        "$default" : null },
    "$$minExclusie" : { "$type" : "atomic",
                        "$default" : null },
    "$$minInclusie" : { "$type" : "atomic",
                        "$default" : null },
    "$$explicitTimezone" : { "$type" : { "$atomic" : "string",
                                         "$enum" : [ "required", "prohibited", "optional" ] },
                             "$optional" : "optional" },
    "$$constraints" : { "$type" : { "$array" : [ "string" ],
                        "$default" : [] },
  }
}
